float sum(float *l, float *r) {
    if (l == r) { return 0; }
    if (l + 1 == r) { return *l; }
    float *m = l + (r - l) / 2;
    return sum(l, m) + sum(m, r);
}

# 3.3. Kahan Т▒ѓтњї

Kahan Т▒ѓтњї (opens new window) Тў»СИђуДЇУАЦтЂ┐уџёТ▒ѓтњїу«ЌТ│ЋсђѓС╗БуаЂтдѓСИІ№╝џ

float sum(const std::vector<float> &a) {
    float res = 0;
    float c = 0;
    for (auto y : a) {
        y -= c;
        float t = res + y;
        c = (t - res) - y;
        res = t;
    }
    return res;
}
